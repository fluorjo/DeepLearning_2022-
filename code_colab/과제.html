model3=Sequential([
    Dense(1024, input_dim=64, activation='relu'),
    Dense(512, activation='relu'),
    Dense(256, activation='relu'),
    Dense(128, activation='relu'),
    Dense(10, activation='softmax')
])
model3.summary()
model3.compile('adam',loss='sparse_categorical_crossentropy',metrics=['accuracy']) 
mc = ModelCheckpoint('digits_best.h5', monitor='val_loss', verbose=0, save_best_only=True)
es = EarlyStopping(patience=50)
hist = model3.fit(X_train, Y_train, epochs=1000, verbose=0, callbacks=[mc, es], validation_split=0.2, batch_size=64)

model3.evaluate(X_test,Y_test)
y_acc=hist.history['accuracy']
y_vloss=hist.history['val_loss']
xs=np.arange(1, len(y_acc)+1)
plt.figure(figsize=(12,8))
plt.plot(xs,y_acc,label='train accuracy')
plt.plot(xs,y_vloss,label='val_loss')
plt.legend(),
plt.grid(),
plt.xlabel('Epoch')
